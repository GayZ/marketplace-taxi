h2. object ClientAddress

|_.Атрибут|_.[[Тип|Типы JSON атрибутов]]|_.Обязательность|_.Описание|
|address|object Address|обязательный|Адрес|
|entrance|string|опциональный|Подъезд|
|flat|string|опциональный|Квартира|
|comment|string|опциональный|Комментарий|

h2. object Address

|_.Атрибут|_.[[Тип|Типы JSON атрибутов]]|_.Обязательность|_.Описание|
|name|string|обязательный|Форматированное наименование|
|components|array object AddressComponent|опциональный|Компоненты|
|types|object AddressTypes|опциональный|Идентификатор типа|
|position|object GpsPosition|опциональный|Координаты|

h2. object AddressComponent

|_.Атрибут|_.[[Тип|Типы JSON атрибутов]]|_.Обязательность|_.Описание|
|level|number AddressLevel|обязательный|Уровень|
|name|string|обязательный|Наименование|

h2. object AddressTypes

|_.Атрибут|_.[[Тип|Типы JSON атрибутов]]|_.Обязательность|_.Описание|
|pointType|number|опциональный|Тип точки|
|aliasType|number|опциональный|Тип быстрого адреса|

h2. object GpsPosition

|_.Атрибут|_.[[Тип|Типы JSON атрибутов]]|_.Обязательность|_.Описание|
|lat|number|обязательный|Широта|
|lon|number|обязательный|Долгота|

h2. number AddressLevel

|_.Значение|_.Описание|
|0|Страна|
|1|Административный уровень 1|
|2|Административный уровень 2|
|3|Административный уровень 3|
|4|Административный уровень 4|
|5|Административный уровень 5|
|6|Административный уровень 6|
|7|Улица|
|8|Дом|
|9|Быстрый адрес|
